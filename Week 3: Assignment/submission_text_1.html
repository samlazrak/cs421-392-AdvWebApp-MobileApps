/* (15 Points, CS392) Implement a function that calculates the remainder of a given dividend and divisor  
without using the modulus (%) operator.  
You can assume the dividend and divisor are both positive, non-zero values. */
function Remainder(dividend, divisor) {
    for (i = 0; i < (dividend / divisor); i++) {
        var temp = (dividend - (divisor * i));
    }
    return temp;
}
/* (10 Points) Implement a function that accepts an array of numbers and returns the sum of all the numbers in the list.

function sum(total, n) {
