import turtle

odd_number = 361
double_number = 1.5
sides = 25
length = 40

def drawShape(the_turtle, sides, length):
for i in range(0, sides):
the_turtle.forward(length)
the_turtle.left(odd_number/sides)
def draw():
turtleScreen = turtle.Screen()
the_turtle = turtle.Turtle()
the_turtle.speed(0)

for i in range(0, odd_number):
drawShape(the_turtle, sides, length)
the_turtle.left(odd_number/double_number)

draw()
